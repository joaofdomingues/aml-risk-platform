Core SQL · Risk Scoring Logic
-- AML Risk Score: composite weighted model WITH customer_risk AS ( SELECT t.customer_id, SUM(t.amount) AS total_30d, COUNT(*) AS tx_count, COUNT(DISTINCT t.country_code) AS country_spread, c.pep_flag, c.kyc_tier FROM fact_transactions t JOIN dim_customer c ON t.customer_id = c.id WHERE t.tx_date >= DATEADD(day, -30, GETDATE()) GROUP BY t.customer_id, c.pep_flag, c.kyc_tier ), risk_scored AS ( SELECT *, -- Weighted composite risk score (0–100) (CASE WHEN total_30d > 500000 THEN 40 ELSE total_30d/12500 END + CASE WHEN country_spread > 5 THEN 25 ELSE country_spread*5 END + CASE WHEN pep_flag = 1 THEN 20 ELSE 0 END + CASE WHEN kyc_tier = 'LOW' THEN 15 ELSE 0 END) AS risk_score FROM customer_risk ) SELECT *, CASE WHEN risk_score >= 70 THEN 'HIGH' WHEN risk_score >= 40 THEN 'MEDIUM' ELSE 'LOW' END AS risk_tier FROM risk_scored ORDER BY risk_score DESC;
